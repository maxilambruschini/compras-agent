Imports: argparse, asyncio, base64, json, os, sys; from decimal import Decimal; from pathlib import Path; from typing import Any; from anthropic import Anthropic; from openai import AsyncOpenAI. From app: get_settings; from app.models.extraction import ExtractedInvoice; from app.services.extraction import ExtractionService, SYSTEM_PROMPT; from app.services.storage import LocalStorageBackend.

       - FIXTURE_NAMES = ("factura_a", "remito", "lista_informal")
       - REPORT_PATH = FIXTURE_DIR / "calibration_report.json"
       - CLAUDE_MODEL = "claude-opus-4-7"   # D-12 — NEVER gpt-4o for ground truth
       - GPT_MODEL = "gpt-4o"
       - SUPPORTED_EXTS = (".jpg", ".jpeg", ".png")

    4. def require_env(name: str) -> str — returns os.environ[name] or prints "missing env: {name}" to stderr and sys.exit(2). MUST NOT print the value.

    5. def detect_media_type(path: Path) -> str — maps .jpg/.jpeg → "image/jpeg", .png → "image/png", default "image/jpeg".

    6. def find_fixture(name: str) -> Path | None — searches FIXTURE_DIR for {name}.jpg, .jpeg, .png; returns first match or None.

    7. def generate_ground_truth(fixture_path: Path, schema_json_str: str) -> dict — verbatim Pattern 8 from RESEARCH.md. Reads bytes, base64-encodes, calls `Anthropic().messages.create(model=CLAUDE_MODEL, max_tokens=4096, messages=[...])` with a single user message containing a text block (instructions + embedded schema_json_str) and an image block (base64 source). Extracts response.content[0].text; strips leading/trailing whitespace + ```json fences if present; returns json.loads(text).

    8. async def run_gpt4o_extraction(fixture_path: Path, settings) -> dict — uses a tempfile-based directory for LocalStorageBackend root (so calibration does NOT write into /data/invoices); constructs AsyncOpenAI(api_key=settings.openai_api_key); constructs ExtractionService; calls await service.extract(fixture_path.read_bytes(), fixture_path.name); returns result.invoice.model_dump(mode="json").

    9. def normalize_value(v: Any) -> Any — recursive normalization: convert Decimal-shaped strings ("1.00" vs "1.0") to a single canonical form via Decimal(str(v)) when v matches r"^-?\d+(\.\d+)?$"; strip surrounding whitespace from strings; convert "" to None; pass through dicts/lists element-wise.

    10. def diff_invoice(ground_truth: dict, extracted: dict, path: str = "") -> list[str] — recursive field-by-field diff returning a list of human-readable strings like "tipo_comprobante: ground_truth=FACTURA_A, extracted=FACTURA_B". For line_items: if lengths differ, emit one diff entry "line_items: len(gt)=N len(ext)=M"; else diff by index recursively. Use normalize_value before equality check.

    11. def cmd_generate_ground_truth(args) -> int:
        - require_env("ANTHROPIC_API_KEY")
        - schema_json = json.dumps(ExtractedInvoice.model_json_schema(), indent=2, ensure_ascii=False)
        - names = [args.fixture] if args.fixture else list(FIXTURE_NAMES)
        - For each name: path = find_fixture(name); if path is None → print "[SKIP] {name}: no image file found in {FIXTURE_DIR}" and continue.
        - Otherwise: data = generate_ground_truth(path, schema_json); out = FIXTURE_DIR / f"{name}_ground_truth.json"; out.write_text(json.dumps(data, indent=2, ensure_ascii=False)); print "[OK] wrote {out}".
        - Return 0.

    12. def cmd_diff(args) -> int:
        - require_env("OPENAI_API_KEY")
        - settings = get_settings()
        - report = {"system_prompt": SYSTEM_PROMPT, "fixtures": {}, "clean": True, "checked": 0}
        - names = [args.fixture] if args.fixture else list(FIXTURE_NAMES)
        - For each name:
          - path = find_fixture(name); if None → report["fixtures"][name] = {"status": "skip_no_image"}; continue.
          - gt_path = FIXTURE_DIR / f"{name}_ground_truth.json"; if not gt_path.exists() → report["fixtures"][name] = {"status": "skip_no_ground_truth", "hint": "run --generate-ground-truth first"}; report["clean"] = False; continue.
          - extracted = asyncio.run(run_gpt4o_extraction(path, settings))
          - ground_truth = json.loads(gt_path.read_text())
          - diffs = diff_invoice(ground_truth, extracted)
          - report["checked"] += 1
          - report["fixtures"][name] = {"status": "checked", "diff_count": len(diffs), "diffs": diffs}
          - if diffs: report["clean"] = False
          - print [OK]/[DIFF N] line summary; print full diff list when N > 0.
        - REPORT_PATH.write_text(json.dumps(report, indent=2, ensure_ascii=False))
        - print "calibration_report.json: {REPORT_PATH}"
        - if report["checked"] == 0: print "no fixtures checked"; return 2.
        - return 0 if report["clean"] else 1.

    13. def main():
        - argparse: parser with three flags: --generate-ground-truth (store_true), --fixture (str, default=None, choices=FIXTURE_NAMES), --report-only (store_true). The --report-only flag currently behaves identically to no flag (diff loop) — it documents intent and reserves a hook for skipping Anthropic; current diff loop already skips Anthropic.
        - If args.generate_ground_truth: sys.exit(cmd_generate_ground_truth(args)); else sys.exit(cmd_diff(args)).

    14. if __name__ == "__main__": main()

    15. Path-safety: NEVER print or log raw API keys. The script may print fixture names, diff counts, diff field paths, and counts only. No --debug flag.
